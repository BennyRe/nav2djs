  5 NAV2D.Navigator = function(options) {
  6   var that = this;
  7   var options = options || {};
  8   var ros = options.ros;
  9   var serverName = options.serverName || '/move_base';
 10   var actionName = options.actionName || 'move_base_msgs/MoveBaseAction';
 11   this.rootObject = options.rootObject || new createjs.Container();
 12 
 13   // get a handle to the stage
 14   if (that.rootObject instanceof createjs.Stage) {
 15     var stage = that.rootObject;
 16   } else {
 17     var stage = that.rootObject.getStage();
 18   }
 19 
 20   // marker for the robot
 21   var robotMarker = new ROS2D.NavigationArrow({
 22     size : 12,
 23     strokeSize : 1,
 24     fillColor : createjs.Graphics.getRGB(255, 128, 0, 0.66)
 25   });
 26   // wait for a pose to come in first
 27   robotMarker.visible = false;
 28   this.rootObject.addChild(robotMarker);
 29 
 30   // setup a listener for the robot pose
 31   var poseListener = new ROSLIB.Topic({
 32     ros : ros,
 33     name : '/robot_pose',
 34     messageType : 'geometry_msgs/Pose',
 35     throttle_rate : 100
 36   });
 37   poseListener.subscribe(function(pose) {
 38     // update the robots position on the map
 39     robotMarker.x = pose.position.x;
 40     robotMarker.y = -pose.position.y;
 41     robotMarker.scaleX = 1.0 / stage.scaleX;
 42     robotMarker.scaleY = 1.0 / stage.scaleY;
 43 
 44     // change the angle
 45     robotMarker.rotation = stage.rosQuaternionToGlobalTheta(pose.orientation);
 46 
 47     robotMarker.visible = true;
 48   });
 49 
 50   // setup a double click listener (no orientation)
 51   this.rootObject.addEventListener('dblclick', function(event) {
 52     // convert to ROS coordinates
 53     var coords = stage.globalToRos(event.stageX, event.stageY);
 54     console.log(coords);
 55   });
 56 };
